\textbf{Aufgabe 1: } \emph{Arithmetische Reihe}
\begin{align*}
&L_n=\pi r+\pi e(n-1),\ \ \ \ n=1,2,3,\dots \\
&\Rightarrow\hspace{0.3cm}S_n=n\pi r+\pi e\sum\limits_{k=1}^n(k-1)=\pi n\left(r+e\frac{n-1}{2}\right)
\end{align*}\\[0.5cm]
